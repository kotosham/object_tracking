# ML / open-vocab detector dependencies for object_tracking (edge GPU side).
#
# Install into a venv created WITH access to the system ROS 2 packages so the
# tracker node can import rclpy / cv_bridge:
#
#   python3 -m venv --system-site-packages ~/ot_venv
#   source ~/ot_venv/bin/activate
#   pip install torch torchvision --index-url https://download.pytorch.org/whl/cu124
#   pip install -r requirements.txt
#   pip install git+https://github.com/ChaoningZhang/MobileSAM.git
#
# CRITICAL pins (Ubuntu 24.04 / ROS 2 Jazzy, verified working set):
#   numpy<2 and scipy<1.14 are REQUIRED. System ROS packages (cv_bridge,
#   matplotlib) are compiled against numpy 1.26 (noble). numpy 2.x breaks the
#   cv_bridge import (C-ABI), and a numpy-2-era scipy references np.long which
#   numpy 1.26 does not have ("module 'numpy' has no attribute 'long'").
#   Keeping the venv on numpy 1.26.4 + scipy 1.13.1 aligns it with ROS.
#
# torch is NOT pinned here — install it from the CUDA wheel index (see above);
# verified with torch 2.6.0+cu124 on a GTX 1650 (CUDA 12.7 driver).
# mobile_sam is not on PyPI — install from the git URL above (pulls timm).
#
# Verified-working versions (2026-06): numpy 1.26.4, scipy 1.13.1,
# ultralytics 8.4.75, transformers 5.12.1, timm (latest), pillow 10.x.

numpy<2
scipy<1.14
ultralytics
transformers
timm
pillow
